        const prepositions = ["of", "to", "in", "from", "about", "for", "with", "at", "on", "against"];

        function startGame() {

            document.getElementById('start-screen').classList.add('hidden');

            document.getElementById('game-screen').classList.remove('hidden');

           

            let shuffled = [...vocabularyDB].sort(() => 0.5 - Math.random());

            activePool = shuffled.slice(0, 20).map(item => ({ ...item, correctCount: 0 }));

 

            startTimer();

            pickWord();

        }

 

        function startTimer() {

            timerInterval = setInterval(() => {

                timeLeft--;

                let mins = Math.floor(timeLeft / 60);

                let secs = timeLeft % 60;

                document.getElementById('timer').innerText = `${mins < 10 ? '0' : ''}${mins}:${secs < 10 ? '0' : ''}${secs}`;

                if (timeLeft <= 0) endGame();

            }, 1000);

        }

 

        function pickWord() {

            isAnswering = false;

            currentWord = activePool[Math.floor(Math.random() * activePool.length)];

           

            document.getElementById('word-display').innerText = currentWord.word;

            document.getElementById('word-meaning').innerText = currentWord.mean;

           

            const prog = (currentWord.correctCount / 3) * 100;

            document.getElementById('word-progress-bar').style.width = prog + "%";

            document.getElementById('word-progress-text').innerText = `${currentWord.correctCount}/3 Lần`;

 

            renderOptions();

        }

 

        function renderOptions() {

            const container = document.getElementById('options');

            container.innerHTML = '';

           

            let choices = [currentWord.prep];

            let otherPreps = prepositions.filter(p => p !== currentWord.prep);

            let shuffledOthers = otherPreps.sort(() => 0.5 - Math.random()).slice(0, 3);

            choices = [...choices, ...shuffledOthers].sort(() => 0.5 - Math.random());

 

            choices.forEach(p => {

                const btn = document.createElement('button');

                btn.className = "option-btn bg-white border-2 border-gray-100 py-3 sm:py-4 rounded-xl text-lg sm:text-xl font-bold text-indigo-800 hover:border-indigo-400";

                btn.innerText = p;

                btn.onclick = () => checkAnswer(p, btn);

                container.appendChild(btn);

            });

        }

 

        function checkAnswer(selected, btn) {

            if (isAnswering) return;

            isAnswering = true;

 

            if (selected === currentWord.prep) {

                score += 2;

                totalCorrect++;

                currentWord.correctCount++;

                btn.classList.add('correct-anim');

            } else {

                score = Math.max(0, score - 1);

                totalWrong++;

                const wordKey = currentWord.word.replace('...', '').trim();

                wrongWordsStats[wordKey] = (wrongWordsStats[wordKey] || 0) + 1;

                btn.classList.add('wrong-anim');

                document.getElementById('wrong-audio').play();

               

                Array.from(document.getElementsByClassName('option-btn')).forEach(b => {

                    if (b.innerText === currentWord.prep) b.classList.add('correct-anim');

                });

            }

 

            document.getElementById('score').innerText = score;

 

            setTimeout(() => {

                if (currentWord.correctCount >= 3) {

                    replaceWordInPool(currentWord);

                }

                pickWord();

            }, 800);

        }

 

        function replaceWordInPool(word) {

            const index = activePool.indexOf(word);

            const poolWords = activePool.map(w => w.word);

            const available = vocabularyDB.filter(w => !poolWords.includes(w.word));

            if (available.length > 0) {

                const newW = available[Math.floor(Math.random() * available.length)];

                activePool[index] = { ...newW, correctCount: 0 };

            } else {

                word.correctCount = 0;

            }

        }

 

        function endGame() {

            clearInterval(timerInterval);

            document.getElementById('game-screen').classList.add('hidden');

            document.getElementById('result-screen').classList.remove('hidden');

           

            document.getElementById('final-score').innerText = score;

            document.getElementById('stat-correct').innerText = totalCorrect;

            document.getElementById('stat-wrong').innerText = totalWrong;

 

            const wrongList = document.getElementById('wrong-words-list');

            const sortedWrong = Object.entries(wrongWordsStats).filter(entry => entry[1] >= 1).sort((a,b) => b[1] - a[1]);

           

            if (sortedWrong.length > 0) {

                document.getElementById('wrong-words-container').classList.remove('hidden');

                wrongList.innerHTML = sortedWrong.map(item => `<div>• <b>${item[0]}</b>: sai ${item[1]} lần</div>`).join('');

            }

        }

                alert("VUI LÒNG NHẬP ĐẦY ĐỦ THÔNG TIN!");

                return;

            }

 

            let ranks = JSON.parse(localStorage.getItem('english_ranks_u5_responsive') || '[]');

            ranks.push({ name, class: className, score, date: new Date().getTime() });

            ranks.sort((a, b) => b.score - a.score);

            localStorage.setItem('english_ranks_u5_responsive', JSON.stringify(ranks));

 

            displayLeaderboard(ranks);

        }

 

        function displayLeaderboard(ranks) {

            document.getElementById('rank-form').classList.add('hidden');

            document.getElementById('leaderboard').classList.remove('hidden');

            const list = document.getElementById('rank-list');

            list.innerHTML = '';

           

            ranks.slice(0, 10).forEach((r, i) => {

                const item = document.createElement('div');

                item.className = "rank-item flex justify-between py-2 border-b border-gray-50 uppercase";

                item.innerHTML = `<span class="truncate pr-2">#${i+1} ${r.name} - ${r.class}</span> <span class="font-black text-indigo-600 shrink-0">${r.score}đ</span>`;

                list.appendChild(item);

            });

        }
